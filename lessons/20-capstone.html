from llama_index.core import (
    SimpleDirectoryReader, VectorStoreIndex, StorageContext,
    load_index_from_storage, Settings, get_response_synthesizer,
)
from llama_index.core.node_parser import SentenceSplitter
from llama_index.core.ingestion import IngestionPipeline
from llama_index.core.postprocessor import SimilarityPostprocessor
from llama_index.core.query_engine import RetrieverQueryEngine
from llama_index.core.evaluation import FaithfulnessEvaluator
from llama_index.llms.openai import OpenAI
from llama_index.embeddings.openai import OpenAIEmbedding
import os

Settings.llm = OpenAI(model='gpt-4o-mini')
Settings.embed_model = OpenAIEmbedding(model='text-embedding-3-small')

PERSIST = './storage'
if os.path.exists(PERSIST):                      # 已建过就秒加载
    index = load_index_from_storage(StorageContext.from_defaults(persist_dir=PERSIST))
else:                                            # 首次：摄取 -> 建索引 -> 落盘
    docs = SimpleDirectoryReader('./data').load_data()
    nodes = IngestionPipeline(transformations=[SentenceSplitter(chunk_size=512, chunk_overlap=50)]).run(documents=docs)
    index = VectorStoreIndex(nodes)
    index.storage_context.persist(persist_dir=PERSIST)

engine = RetrieverQueryEngine.from_args(
    retriever=index.as_retriever(similarity_top_k=4),
    node_postprocessors=[SimilarityPostprocessor(similarity_cutoff=0.7)],
    response_synthesizer=get_response_synthesizer(response_mode='compact'),
)
resp = engine.query('退款政策是什么？')
print('答案:', resp)
print('忠实度:', FaithfulnessEvaluator().evaluate_response(response=resp).passing)

chat = index.as_chat_engine(chat_mode='condense_plus_context')
print(chat.chat('那国际订单也一样吗？'))
✅ 本课要点✅ Key points
  • 完整 RAG = 写入路径（建一次、落盘）+ 查询路径（反复问）。A full RAG = write path (build once, persist) + query path (ask repeatedly).
  • 每一站都是可替换组件：换库、换检索、换合成都不动主结构。Every stop is a swappable component: change store/retrieval/synthesis without touching the skeleton.
  • 持久化 + 评估让它从 demo 走向可维护、可回归。Persistence + evaluation take it from demo to maintainable and regression-safe.
💡 设计亮点💡 Design insight
整条管道每一站都能独立替换，却共享统一接口——这正是 LlamaIndex 的核心思想：用可组合的标准件搭出你自己的 RAG。
Every stop swaps independently yet shares one interface — that's LlamaIndex's core idea: compose your own RAG from standard, interchangeable parts.
